public class Linked {

    /**
     * 递归
	 * 将当前操作的节点,反转并 将其的下一个节点值设置为null
	 * 以下为数据的正序
	 * 0 1 2 3 null
	 * 以下是经过递归处理数据的逻辑
	 * null 和 3 不处理 ,因为 3 和 null 本身 或者  下一个节点是null
	 * 2 获取下一个节点  3 ,给 3设置 下一个节点 2 ,2 的下一个节点为 null
	 * 1 获取下一个节点 2 ,给 2设置 下一个节点 1 ,1 的下一个节点为 null ,一次递推,计算出结果
	 * 3 2 1 0 null
     * @param head
     * @return
     */
    private static Node Reverse1(Node head) {
        if (head == null || head.getNext() == null) {
            return head;
        }
        Node reNode = Reverse1(head.getNext());
        head.getNext().setNext(head);
        head.setNext(null);
        return reNode;
	}

	/**
	 * 遍历反转
	 * 递归反转法是从后往前逆序反转指针域的指向，而遍历反转法是从前往后反转各个结点的指针域的指向。
	 * 基本思路是：
	 * 1. 缓存 当前节点 cur 的下一个节点
	 * 2. 更改当前节点指针指向上一结点 pre cur.setNext(pre) 。
	 * 3. 把当前结点的指针域用 tmp 临时保存，以便下一次使用
	 * 4. 循环介绍将头节点 指针指向null head.setNext(null);
	 *    pre：上一结点
	 *    cur: 当前结点
	 *    tmp: 临时结点，用于保存当前结点的指针域（即下一结点）
	 * @param head
	 * @return
	 */
   private static Node Reverse(Node head) {
       if (head == null || head.getNext() == null ) {
		  return head;
	   }
		Node pre = head;
		Node cur = head.getNext();
		Node tmp;
		while (cur != null){
			tmp = cur.getNext();
			cur.setNext(pre);

			// 指针往下移动
			pre = cur;
			cur = tmp;
		}
		// 最后将原链表的头节点的指针域置为null
		head.setNext(null);
		return pre;

	}

	public static void main(String[] args) {
		Node head = new Node(0);
		Node node1 = new Node(1);
		Node node2 = new Node(2);
		Node node3 = new Node(3);
		head.setNext(node1);
		node1.setNext(node2);
		node2.setNext(node3);


		System.out.println("*************反转前***************");
        print(head);

		System.out.println("*************遍历反转*************");
        head = Reverse(head);
		 print(head);

		System.out.println("*************递归反转*************");
		head = Reverse1(head);
        print(head);

	}
    private static void print(Node head) {
        while (null != head) {
            System.out.print(head.getData() + " ");
            head = head.getNext();
        }
		System.out.println();
   }

}
class Node {
	private int Data;// 数据域
	private Node Next;// 指针域
	Node(int Data) {
   		this.Data = Data;
	}
	int getData() {
		return Data;
	}
	public void setData(int Data) {
		this.Data = Data;
	}

	Node getNext() {
		return Next;
	}
	void setNext(Node Next) {
		this.Next = Next;
	}

	@Override
	public String toString() {
		return "Node{" +
				"Data=" + Data+'}';
	}
}
