public class Sort {
    public static void main(String[] args) {
        int[] arr = {9, 1, 3, 5, 6, 4, 2};
        bubbleSort(arr);
        insertSort(arr);
        selectSort(arr);
    }
 
    /**
     * 冒泡排序
     * 思路:
     * j  和  j+1 比较
     * i 标识第几次排序
     * arr.length - 1 - i  标识未排好序的最大下标
     *
     * @param arr
     */
    private static void bubbleSort(int[] arr) {
        int temp;
        for (int i = 0; i < arr.length; i++) {
            for (int j = 0; j < arr.length - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    temp = arr[j + 1];
                    arr[j + 1] = arr[j];
                    arr[j] = temp;
                }
            }
        }
        print(arr);
    }

    /**
     * 插入排序
     * 思路:
     * i  是 需要排序的长度
     *
     * @param arr
     */
    private static void insertSort(int[] arr) {
        int temp;
        for (int i = 1; i < arr.length; i++) {
            for (int j = 0; j < i; j++) {
                if (arr[j] > arr[i]) {
                    temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
        }
        print(arr);
    }

    /**
     * 选择排序
     * 思路:
     * i : 当前需要排序的位置
     * j : 迭代变量
     * 每次循环,找出当前最小值下标, 并交换
     *
     * @param arr
     */
    private static void selectSort(int[] arr) {
        int temp, minIndex;
        for (int i = 0; i < arr.length; i++) {
            minIndex = i;
            for (int j = i; j < arr.length; j++) {
                if (arr[minIndex] > arr[j]) {
                    minIndex = arr[j];
                }
            }
            temp = arr[minIndex];
            arr[minIndex] = arr[i];
            arr[i] = temp;

        }
        print(arr);
    }
    
    private static void print(int[] arr1) {
        for (int ints : arr1) {
            System.out.print(ints);
            System.out.print(",");
        }
        System.out.println();
    }




}
